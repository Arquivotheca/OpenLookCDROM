% Definition for bitmap images to be included 
\def\sunbitmap#1#2#3#4{%
 \hfill \hbox to #1{%
  \vbox to #2{
   \vss
   \special{pre: sunbitmap #3 #40}
   \special{ps: sunbitmap #3 #40}
  }%
 }%
 \hfill
}
